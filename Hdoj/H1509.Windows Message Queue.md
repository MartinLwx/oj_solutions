#include<bits/stdc++.h>
using namespace std;
struct node {
    string msg;
    int para, priority, idx;
};
vector<node> v;
bool cmp(node& x, node& y) {
    if(x.priority == y. priority)
        return x.idx > y.idx;
    return x.priority > y.priority;
}
int main() {
    string s;
    int idx = 1;
    while(cin >> s) {
        if(s == "GET") {
            if(v.size() == 0) {
